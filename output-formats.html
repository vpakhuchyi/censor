Output Formats

Censor supports two primary output formats: Text and JSON. Each format is designed for specific use cases and provides different formatting options.

Overview​

Text Format​

  • Human-readable output
  • Default format for direct printing and logging
  • Ideal for development and debugging
  • Supports custom formatting patterns

JSON Format​

  • Machine-readable output
  • Suitable for structured logging and APIs
  • Preserves data types
  • Compatible with JSON parsers

Text Format​

The text format is designed for human readability and is the default output format.

Basic Usage​

package main

import (
    "fmt"
    "github.com/vpakhuchyi/censor"
)

func main() {
    c := censor.New()

    type User struct {
        ID       string `censor:"display"`
        Email    string
        Password string
    }

    user := User{
        ID:       "123",
        Email:    "user@example.com",
        Password: "secret",
    }

    // Text format output
    masked := c.Process(user)
    fmt.Printf("%v\n", masked)
    // Output: {ID: 123, Email: [CENSORED], Password: [CENSORED]}
}

Text Format Features​

  1. Struct Formatting

    type Address struct {
        Street string
        City   string
    }

    type User struct {
        ID      string  `censor:"display"`
        Address Address
    }

    user := User{
        ID: "123",
        Address: Address{
            Street: "123 Main St",
            City:   "Anytown",
        },
    }

    fmt.Printf("%v\n", c.Process(user))
    // Output: {ID: 123, Address: {Street: [CENSORED], City: [CENSORED]}}

  2. Slice and Array Formatting

    data := []string{"secret1", "secret2"}
    fmt.Printf("%v\n", c.Process(data))
    // Output: [[CENSORED], [CENSORED]]

  3. Map Formatting

    data := map[string]string{
        "visible": "public",
        "hidden":  "secret",
    }
    fmt.Printf("%v\n", c.Process(data))
    // Output: map[visible: public, hidden: [CENSORED]]

JSON Format​

The JSON format is designed for machine readability and structured data interchange.

Basic Usage​

package main

import (
    "encoding/json"
    "fmt"
    "github.com/vpakhuchyi/censor"
)

func main() {
    c := censor.New()

    type User struct {
        ID       string `json:"id" censor:"display"`
        Email    string `json:"email"`
        Password string `json:"password"`
    }

    user := User{
        ID:       "123",
        Email:    "user@example.com",
        Password: "secret",
    }

    // Process and convert to JSON
    masked := c.Process(user)
    jsonData, _ := json.Marshal(masked)
    fmt.Println(string(jsonData))
    // Output: {"id":"123","email":"[CENSORED]","password":"[CENSORED]"}
}

JSON Format Features​

  1. Struct to JSON

    type Address struct {
        Street string `json:"street"`
        City   string `json:"city"`
    }

    type User struct {
        ID      string  `json:"id" censor:"display"`
        Address Address `json:"address"`
    }

    user := User{
        ID: "123",
        Address: Address{
            Street: "123 Main St",
            City:   "Anytown",
        },
    }

    masked := c.Process(user)
    jsonData, _ := json.MarshalIndent(masked, "", "  ")
    fmt.Println(string(jsonData))
    // Output:
    // {
    //   "id": "123",
    //   "address": {
    //     "street": "[CENSORED]",
    //     "city": "[CENSORED]"
    //   }
    // }

  2. Arrays and Slices in JSON

    type User struct {
        ID     string   `json:"id" censor:"display"`
        Emails []string `json:"emails"`
    }

    user := User{
        ID:     "123",
        Emails: []string{"user1@example.com", "user2@example.com"},
    }

    masked := c.Process(user)
    jsonData, _ := json.Marshal(masked)
    fmt.Println(string(jsonData))
    // Output: {"id":"123","emails":["[CENSORED]","[CENSORED]"]}

  3. Maps in JSON

    data := map[string]interface{}{
        "id": "123",
        "credentials": map[string]string{
            "username": "john_doe",
            "password": "secret",
        },
    }

    masked := c.Process(data)
    jsonData, _ := json.MarshalIndent(masked, "", "  ")
    fmt.Println(string(jsonData))
    // Output:
    // {
    //   "id": "123",
    //   "credentials": {
    //     "username": "john_doe",
    //     "password": "[CENSORED]"
    //   }
    // }

Choosing the Right Format​

Use Text Format When:​

  • Debugging application behavior
  • Reading logs directly
  • Printing to console
  • Need human-readable output
  • Working with standard Go printing functions

Use JSON Format When:​

  • Integrating with logging systems
  • Building APIs
  • Storing structured data
  • Need machine-readable output
  • Working with JSON-based tools and services

Format-Specific Configuration​

Text Format Configuration​

cfg := censor.Config{
    Formatter: censor.FormatterConfig{
        MaskValue: "***",            // Custom mask for text output
        ExcludePatterns: []string{
            `\d{4}-\d{4}-\d{4}-\d{4}`,
        },
    },
}

c := censor.NewWithOpts(censor.WithConfig(&cfg))

JSON Format Configuration​

cfg := censor.Config{
    Formatter: censor.FormatterConfig{
        MaskValue: "\"[MASKED]\"",  // Custom mask for JSON output
        ExcludePatterns: []string{
            `\d{4}-\d{4}-\d{4}-\d{4}`,
        },
    },
}

c := censor.NewWithOpts(censor.WithConfig(&cfg))

Best Practices​

  1. Consistent Format Usage

    // Good: Consistent text format
    logger.Printf("User: %v", c.Process(user))
    logger.Printf("Payment: %v", c.Process(payment))

    // Good: Consistent JSON format
    jsonLogger.Log(c.Process(user))
    jsonLogger.Log(c.Process(payment))

    // Bad: Mixed formats
    logger.Printf("User: %v", c.Process(user))
    jsonLogger.Log(c.Process(user))

  2. Format-Specific Masking

    // Text format configuration
    textCfg := censor.Config{
        Formatter: censor.FormatterConfig{
            MaskValue: "***",
        },
    }

    // JSON format configuration
    jsonCfg := censor.Config{
        Formatter: censor.FormatterConfig{
            MaskValue: "\"[MASKED]\"",
        },
    }

  3. Logger Integration

    // Text format for standard logging
    stdLogger := log.New(os.Stdout, "", log.LstdFlags)
    stdLogger.Printf("Data: %v", c.Process(data))

    // JSON format for structured logging
    jsonLogger := zerolog.New(os.Stdout)
    jsonLogger.Info().Interface("data", c.Process(data)).Send()

Common Pitfalls​

  1. Mixing Formats

    // Problem: Inconsistent output format
    data := map[string]string{"password": "secret"}

    fmt.Printf("%v\n", c.Process(data))        // Text format
    json.NewEncoder(w).Encode(c.Process(data)) // JSON format

    // Solution: Use consistent format for each context
    textLogger := log.New(os.Stdout, "", 0)
    jsonLogger := zerolog.New(os.Stdout)

    textLogger.Printf("%v", c.Process(data))   // Always text
    jsonLogger.Info().Interface("data", c.Process(data)).Send() // Always JSON

  2. Format-Specific Escaping

    // Problem: Incorrect escaping
    cfg := censor.Config{
        Formatter: censor.FormatterConfig{
            MaskValue: "[CENSORED]", // Might cause JSON parsing issues
        },
    }

    // Solution: Use format-specific masks
    textCfg := censor.Config{
        Formatter: censor.FormatterConfig{
            MaskValue: "[CENSORED]",
        },
    }

    jsonCfg := censor.Config{
        Formatter: censor.FormatterConfig{
            MaskValue: "\"[CENSORED]\"",
        },
    }

  3. Type Preservation

    // Problem: Type mismatch in JSON
    type User struct {
        Age int `json:"age"`
    }

    // Solution: Use appropriate types for each format
    textAge := fmt.Sprintf("%v", c.Process(user.Age))
    jsonAge := c.Process(user.Age).(int)
